setSplitString

최소지원버전
1.0.0.0
지원환경
멀티 모듈, IE 모듈 지원
형식
object.setSplitString(str);
파라미터
이름 형식 설명
str String 구분자 문자열, 기본값은 "[SPLT]"입니다.
반환 값
설명

구분자 문자열을 설정합니다.

DEXTUploadX5는 서버로 메타데이터를 보낼 때, 구분자 문자열을 포함한 문자열 데이터를 전송합니다.

function onDX5Created(id) {
  var dx = dx5.get(id);
  // 구분자용 문자를 파이프라인 문자(|)로 설정합니다.
  dx.setSplitString("|");
  // 구분자용 문자를 출력합니다.
  console.log(dx.getSplitString());
}
