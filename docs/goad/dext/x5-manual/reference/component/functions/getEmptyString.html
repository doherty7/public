getEmptyString

최소지원버전
1.0.0.0
지원환경
멀티 모듈, IE 모듈 지원
형식
splitter = object.getEmptyString();
파라미터
반환 값
공백용 문자열, 기본값은 "[EMPTY]"입니다.
설명

공백을 대표하는 문자열을 반환합니다.

현재 사용되는 곳은 없습니다.

function onDX5Created(id) {
  var dx = dx5.get(id);
  // 공백용 문자를 파이프라인 문자([e])로 설정합니다.
  dx.setEmptyString("[e]");
  // 공백용 문자를 출력합니다.
  console.log(dx.getEmptyString());
}
