downloadById

최소지원버전
1.0.0.0
지원환경
멀티 모듈, IE 모듈 지원
형식
object.downloadById(id);
파라미터
이름 형식 설명
id String 다운로드할 가상 파일의 고유 아이디
반환 값
설명

주어진 고유 아이디에 해당하는 가상 파일을 다운로드합니다.

파일을 다운로드하려면 서버에 존재하는 대상의 경로 정보를 얻을 수 있어야 하는데, 가상 파일의 속성을 사용합니다.

 

멀티 모듈은 다운로드는 하려는 가상 파일은 반드시 downUrl 속성에 다운로드할 수 있는 서버 측 주소가 설정되어야 합니다.

<script>
  function onDX5Created(id) {
    var dx = dx5.get(id);
    // 다운로드 경로 정보가 없어서 다운로드할 수 없는 가상 파일입니다.
    dx.addVirtualFile({ vindex: "IDX0001", name: "가상파일.txt", size: 12345 });
    // 다운로드 경로가 있으면 다운로드할 수 있습니다.
    dx.addVirtualFile({ vindex: "IDX0003", name: "서강대교_509147.jpg", size: 509147, downUrl: "http://../common-download.do?key=FID0001" });
  }

  function download(id) {
    var dx = dx5.get(id);
    // 두 번째 가상 파일을 선택합니다.
    var itemId = dx.getItemId(1);
    dx.downloadById(itemId);
  }
</script>
<button type="button" onclick="download('컴포넌트아이디');">다운로드</button>

멀티 모듈일 때, 다운로드 경로는 서버 측 컴포넌트를 사용하여, 반드시 Content-disposition 응답 헤더 값이 attachment 상태이어야 합니다.

더불어 멀티 모듈은 대상의 존재 유무를 판단하기 위해서 HEAD 메소드 요청을 보내며, HTTP 응답 코드가 오류가 아닌 경우에 다시 다운로드를 수행합니다.

 

IE 모듈에서 가상 파일은 반드시 url 속성에 다운로드할 수 있는 서버 측 주소가 설정되어야 합니다.

<script>
  function onDX5Created(id) {
    var dx = dx5.get(id);
    // 다운로드 경로 정보가 없어서 다운로드할 수 없는 가상 파일입니다.
    dx.addVirtualFile({ vindex: "IDX0001", name: "가상파일.txt", size: 12345 });
    // 다운로드 경로가 있으면 다운로드할 수 있습니다.
    dx.addVirtualFile({ vindex: "IDX0003", name: "서강대교_509147.jpg", size: 509147, url: "http://../common-download.do?key=FID0001" });
  }

  function download(id) {
    var dx = dx5.get(id);
    // 두 번째 가상 파일을 선택합니다.
    var itemId = dx.getItemId(1);
    dx.downloadById(itemId);
  }
</script>
<button type="button" onclick="download('컴포넌트아이디');">다운로드</button>

IE 모듈은 Content-disposition 응답 헤더가 attachement 혹은 inline 인 경우와 상관없이 강제로 파일을 다운로드할 수 있습니다.
