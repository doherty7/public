hasUploadableItems

최소지원버전
1.0.0.0
지원환경
멀티 모듈, IE 모듈 지원
형식
has = object.hasUploadableItems();
파라미터
반환 값
업로드할 수 있는 대상이 있다면 true, 아니면 false
설명

업로드할 수 있는 대상이 있는지를 반환합니다.

function upload(id) {
  var dx = dx5.get(id);
  // 업로드할 대상이 있는 지 확인 후, 업로드를 시작합니다.
  if (dx.hasUploadableItems()) {
    // 업로드를 시작합니다.
  } else {
    alert("업로드할 대상이 없습니다.");
  }
}
