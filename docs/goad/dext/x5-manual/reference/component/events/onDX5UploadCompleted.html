설명

onDX5UploadCompleted 함수는 업로드가 완료될 때 호출되는 콜백 함수입니다.

서버에서 보낸 응답 데이터를 확인하려면 이 함수가 호출된 이후에 확인하면 됩니다.

function onDX5UploadCompleted(id) {
  var result = "";

  // ORAF 방식이라면
  result = dx5.get(id).getResponses(0);

  // OROF 또는 EXNJ 방식이라면
  result = dx5.get(id).getResponses().join("\n");

  alert("업로드가 완료되었습니다.\n응답 데이터는 다음과 같습니다.\n" + result);
}
