get

최소지원버전
1.0.0.0
지원환경
멀티 모듈, IE 모듈 모두 지원
형식
component = dx5.get(id);
파라미터
이름 형식 설명
id String

DEXTUploadX5 컴포넌트의 아이디입니다.

컴넌트가 생성될 때, 사용되는 object 엘리먼트의 id 속성 값이며, dx5 객체는 이 아이디를 가지고 컴포넌트를 구분합니다.
설명

생성된 DEXTUploadX5 컴포넌트를 사용하기 위해서는 dx5.get 함수를 사용하여 컴포넌트 객체를 얻어야만 합니다.

create 함수를 사용하여 생성했다 하더라도, 컴포넌트 생성을 보장받으려면 onDX5Created 콜백 함수가 호출된 이후에 get함수를 사용해야 합니다.

onDX5Created 함수는 컴포넌트가 생성될 때마다 호출됩니다.

<div id="dext5-container" style="width:500px; height:300px;"><!-- 컴포넌트를 담는 컨테이너 요소 --></div>
<button id="btn-add-files" type="button">파일 추가</button>
<script>
  dx5.create({
    // mode를 "multi" 또는 "ie"로 설정하며, id 속성으로 컴포넌트 DOM 아이디를 부여합니다.
    // 이 DOM 아이디는 컴포넌트를 구분 짓는 아이디이기도 합니다.
    // parentId는 컨테이너 아이디를 지정합니다.
    mode: "multi", id: "dext5", parentId: "dext5-container",
    // 컴포넌트는 버튼을 갖고 있지 않으므로 버튼 요소를 직접 만들어주어야 합니다.
    // button, input[type=button] 요소를 만들고 대상 아이디를 지정해주면 파일 추가 기능이 자동으로 연결됩니다.
    btnFile: "btn-add-files"
  });
</script>
<script>
  function onDX5Error(id, code, msg) {
    // 컴포넌트 사용 중에 오류가 발생하면 호출됩니다.
  }

  function onDX5Created(id) {
    // dx5.create 함수를 사용하여 컴포넌트가 만들어지면 onDX5Create 콜백 함수가 호출됩니다.
    // 컴포넌트가 생성되면 dx5.get 함수를 사용하여 컴포넌트를 얻을 수 있습니다.
    var dx = dx5.get(id);
  }
</script>
